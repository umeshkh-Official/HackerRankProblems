# Complete the birthday function below.
def birthday(s, d, m):
    count = 0
    currentSum = 0

    for i in range(0, len(s)):
        currentSum += s[i]

        if i >= m-1:
            if currentSum == d:
                count += 1
            currentSum -= s[i-(m-1)]
            
    return count
